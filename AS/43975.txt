remarks:        Community Meaning
remarks:
remarks:        43975:5002X When advertise to whole Rostelecom (AS12389)
remarks:
remarks:        X=0 - no advertise prefix to upstream
remarks:        X=1 - set one prepend
remarks:        X=2 - set tho prepends
remarks:        X=3 - set three prepends
remarks:        X=4 - set four prepends
remarks:        X=5 - set five prepends
remarks:
remarks:        43975:5LMMN When advertise to Rostelecom (AS12389) from
remarks:        exact Gateways in Samara and N.Novgorod
remarks:
remarks:        L  - Last octet of IP Loopback of Gateway
remarks:        MM - Upstream number on Gateway
remarks:        N  - Community meaning
remarks:
remarks:        N=0 - no advertise prefix to upstream
remarks:        N=1 - set one prepend
remarks:        N=2 - set tho prepends
remarks:        N=3 - set three prepends
remarks:        N=4 - set four prepends
remarks:        N=5 - set five prepends
remarks:
remarks:        43975:5503N - Rostelecom uplink to NNOV-CR1-A on N.Novgorod Gateway1
remarks:        43975:5604N - Rostelecom uplink to NNOV-CR1-B on N.Novgorod Gateway2
remarks:        43975:5705N - Rostelecom uplink to SMRA-CR1-A on Samara Gateway1
remarks:        43975:5802N - Rostelecom uplink to SMRA-CR1-B on Samara Gateway2
